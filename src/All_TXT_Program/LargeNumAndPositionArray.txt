public class LargestElement{

	public static void main(String args[]){

	int arr[]={10,5,4,55,28,18,7};
	
	int largest=arr[0],position=0;

	
	for(int i=1;i<arr.length;i++)
	{
		if(largest<arr[i])
		{
			largest=arr[i];
			position=i;
			
		}
	
	}
	System.out.println("Largest:"+largest +"    Position:"+position);	

}
}
