import java.util.*;
class Main {
    public static void main(String[] args) {
      
      System.out.println("Enter a number for find the palindrome");
      Scanner sc=new Scanner(System.in);
      int num=sc.nextInt();
      int rev=0;
      int origin=num;
      while(num>0)
      {
          rev=rev*10+num%10;
          num=num/10;
      }
      System.out.println(rev);
      
      if(rev==origin)
      {
          System.out.println("this is palendrome");
      }else
        {
            System.out.println("this is not palindome");
        }
      
      
      
       
    }
}
